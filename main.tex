	\begin{frame}
		\frametitle{I.I.D assumption}
		

		
		we use \textbf{independent}
		\begin{align*}
			\theta^\ast &= \arg\max_\theta \prod_{i} q(y_i \vert X_i)  \\
			&= \arg\max_\theta \sum_{i} -y_i \log f_\theta(X_i)
		\end{align*}
		 and \textbf{identically} distributed (i.i.d) samples
		 \begin{align}
		 	X_i,y_i \sim p(X, Y)	
		 \end{align}
		 from a datasets
		 \begin{align}
		 	\{X_i,y_i\}_i^N
		 \end{align}
		
	\end{frame}
